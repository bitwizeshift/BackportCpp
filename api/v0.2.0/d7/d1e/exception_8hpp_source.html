75 // At least on MinGW and Linux, only GCC since 4.7 declares __cxa_get_globals()
76 // in cxxabi.h. Older versions of GCC do not expose this function but it's
77 // there.
78 // On OpenBSD, it seems, the declaration is also missing.
79 // Note that at least on FreeBSD 11, cxxabi.h declares __cxa_get_globals with
80 // a different exception specification, so we can't declare the function
81 // unconditionally. On Linux with clang and libc++ and on OS X, there is a
82 // version of cxxabi.h from libc++abi that doesn't declare __cxa_get_globals,
83 // but provides __cxa_uncaught_exceptions.
84 // The function only appeared in version _LIBCPPABI_VERSION >= 1002 of the
85 // library. Unfortunately, there are linking errors about undefined reference
86 // to __cxa_uncaught_exceptions on Ubuntu Trusty and OS X, so we avoid using
87 // it and forward-declare __cxa_get_globals instead. On QNX SDP 7.0 (QCC 5.4.0),
88 // there are multiple cxxabi.h, one from glibcxx from gcc and another from
89 // libc++abi from LLVM. Which one is included will be determined by the qcc
90 // command line arguments (-V and/or -Y;
91 // http://www.qnx.com/developers/docs/7.0.0/#com.qnx.doc.neutrino.utilities/topic/q/qcc.html
92 // ).
93 // The LLVM libc++abi is missing the declaration of __cxa_get_globals but it is
94 // also patched by QNX developers to not define _LIBCPPABI_VERSION. Older QNX
95 // SDP versions, up to and including 6.6, don't provide LLVM and libc++abi.
96 // See https://github.com/boostorg/core/issues/59.

157 inline BPSTD_INLINE_VISIBILITY
158 int bpstd::uncaught_exceptions()
159  noexcept
160 {
161 #if defined(BPSTD_HAS_UNCAUGHT_EXCEPTIONS)
162  // C++17 implementation
163  return std::uncaught_exceptions();
164 #elif defined(BPSTD_HAS_CXA_GET_GLOBALS)
165  // Tested on {clang 3.2,GCC 3.5.6,GCC 4.1.2,GCC 4.4.6,GCC 4.4.7}x{x32,x64}
166  using byte = unsigned char;
167  auto count = int{};
168  const auto* ptr = reinterpret_cast<const byte*>(::abi::__cxa_get_globals()) + sizeof(void*);
169 
170  // __cxa_eh_globals::uncaughtExceptions, x32 offset - 0x4, x64 - 0x8
171  std::memcpy(&count, ptr, sizeof(count));
172  return count;
173 #elif defined(BPSTD_HAS_GETPTD)
174  // MSVC specific. Tested on {MSVC2005SP1,MSVC2008SP1,MSVC2010SP1,MSVC2012}x{x32,x64}.
175  using byte = unsigned char;
176  auto count = int{};
177 
178  const auto offset = (sizeof(void*) == 8u ? 0x100 : 0x90);
179  const auto* ptr = static_cast<const byte*>(::scope::detail::_getptd()) + offset;
180 
181  // _tiddata::_ProcessingThrow, x32 offset - 0x90, x64 - 0x100
182  std::memcpy(&count, ptr, sizeof(count));
183  return count;
184 #else
185  // Portable C++03 implementation. Does not allow to detect multiple
186  // nested exceptions.
187 
188  // This is a buggy fallback since it will only work with 1 exception
189  // in-flight, but we don't have any other options without exploiting
190  // internal compiler features.
191  return static_cast<int>(std::uncaught_exception());
192 #endif
193 }
